// 사진 확대 기능 const modal = document.getElementById('image-modal'); const modalImage = document.getElementById('modal-image'); const modalClose = document.getElementById('modal-close'); const galleryImages = document.querySelectorAll('.gallery img'); galleryImages.forEach(image => { image.addEventListener('click', () => { modal.style.display = 'flex'; modalImage.src = image.src; }); }); modalClose.addEventListener('click', () => { modal.style.display = 'none'; });
